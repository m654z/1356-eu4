#Major event for Mamlukes to pick weather they will tolerate Copts or pursecute them
country_event = {
	id = flamamluk.3
	title = "flamamluk.3.t"
	desc = "flamamluk.3.d"
	picture = RELIGION_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = MAM
		NOT = { is_year = 1380 }
		is_year = 1357
		religion = sunni
		primary_culture = al_misr_arabic
	}
	
	mean_time_to_happen = {
		months = 30
	}
	
	option = { #Tolerance!
		name = "flamamluk.3.a"
		add_country_modifier = { 
			name = alexandria_tolerated_copts
			duration = 3600
		}
		add_prestige = 2
		ALX = {
			country_event = { 
				id = alexandriaflandy.2
				days = 30
				random = 30
			}
		}

	}
	
	option = { #Protected Minorities no more!
		name = flamamluk.3.b
		360 = {
			change_religion = sunni
			change_culture = al_misr_arabic
		}
		add_prestige = -5
		ALX = {
			country_event = { 
				id = alexandriaflandy.1
				days = 30
				random = 30
			}
		}
	}
}
