# Swedish Nobility Estate Reduction
country_event = {
	id = kalmar.2
	title = "sweden_estate_name"
	desc = "sweden_estate_desc"
	picture = COURT_eventPicture
	fire_only_once = yes
	
	trigger = {
		tag = SWE
		NOT = { is_year = 1456 }
		is_subject = no
	}
	mean_time_to_happen = {
		months = 396
	}
	option = {
		name = "sweden_estate_opt1"
		set_global_flag = sweden_reduced_noble_estates
		IF = {
			limit = { has_dlc = "The Cossacks" }
			add_estate_loyalty = {
				estate = estate_nobles
				loyalty = -20
			}
		}
		IF = {
			limit = { NOT = { has_dlc = "The Cossacks" } }
			every_owned_province = {
				add_unrest = 1
			}
		}
	}
	option = {
		name = "sweden_estate_opt1"
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = 10
		}
		IF = {
			limit = { has_dlc = "The Cossacks" }
			every_owned_province = {
				limit = {
					has_estate = estate_nobles
				}
				add_local_autonomy = 20
			}
		}
		IF = {
			limit = { NOT = { has_dlc = "The Cossacks" }}
			every_owned_province = {
				add_local_autonomy = 10
			}
		}
	}
}
