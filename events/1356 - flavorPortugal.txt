		tag = POR

		name = por_djm.1.a

country_event = { #Event to colonize the Cannaries
	id = por_djm.2
	title = por_djm.2.t
	desc = por_djm.2.d
	picture = EXPLORERS_eventPicture
	
	fire_only_once = yes
	major = yes

	trigger = {
		OR = {
			tag = POR
			tag = CAS
			tag = SPA
			AND = {
				tag = MOR
				NOT = { exists = POR }
				NOT = { exists = CAS }
				NOT = { exists = SPA }
			}
		}
		num_of_ports = 5
		dip_power = 50
		4565 = { NOT = { owned_by = ROOT } }
		366 = { NOT = { owned_by = ROOT } }
	}
	
	mean_time_to_happen = {
		months = 30
		modifier = {
			factor = 0.1
			NOT = { age = age_of_feudalism }
		}
		modifier = {
			factor = 0.1
			is_year = 1390
		}
	}
	
	option = { #We'll colonize it! (POR/CAS/SPA)
		name = por_djm.2.a
		trigger = {
			OR = {
				tag = POR
				tag = CAS
				tag = SPA
			}
		}
		4565 = {
			create_colony = 800
			add_core = ROOT
			change_religion = ROOT
			change_culture = ROOT
		}
		366 = {
			create_colony = 800

			change_religion = ROOT
			change_culture = ROOT
		}
	}
		
	option = { #Sell it to Portugal (CAS)
		name = por_djm_2.b

			tag = CAS
		}
		add_treasury = 100
		4565 = {

		}
