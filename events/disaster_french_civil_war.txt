#Event that gives the nations the option of joining the HRE or getting a claim on all of France
country_event = {
	id = fre_disaster_civil_war.3
	title = fre_disaster_civil_war.3.t
	desc = fre_disaster_civil_war.3.d
	picture = HRE_eventPicture
	
	major = yes
	
	trigger = {
		has_global_flag = 1356_french_civil_war
		NOT = { has_global_flag = emperor_declined_offer_to_french_dukes }
		current_age = age_of_feudalism
		is_subject = no
		culture_group = french
		NOT = { tag = FRA }
		NOT = { tag = ENG }
		NOT = { tag = SPA }
		NOT = { tag = ARA }
		NOT = { tag = NAV }
		NOT = { tag = DAN }
		technology_group = western
	}
	
	mean_time_to_happen = {
		months = 24
	}
	
	option = { #We stay independent!
		name = fre_disaster_civil_war.3.a
		ai_chance = { factor = 95 }
		france_region = {
			add_permanent_claim = ROOT
		}
		add_prestige = 5
		add_legitimacy = 5
		emperor = {
			country_event = { 
			id = fre_disaster_civil_war.6
			days = 15
			}
		}
	}
	
	option = {
		name = fre_disaster_civil_war.3.b
		ai_chance = { factor = 5 }
		ROOT = {
			set_in_empire = yes
		}
		emperor = {
			add_imperial_influence = 2
		
			country_event = { 
				id = fre_disaster_civil_war.5
				days = 15
			}
		}
	}
	
}
